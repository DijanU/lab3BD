Table jugadores {
  id integer [primary key]
  nombre varchar
  posicion varchar
  edad integer
  nacionalidad varchar
  created_at timestamp
}

Table equipos {
  id integer [primary key]
  nombre varchar
  liga varchar
  pais varchar
  fundado integer
}
